func two_choice_options(option1: string, option2: string, result1: string, result2: string): int {
	while true {
	println("");
	println("1. " + option1);
	println("2. " + option2);

	var input = int(input("Choose an option: "));
	
	var output = switch_block_options(false, input, result1, result2, "");
	if output > -1 { return output; }
	}
}

	while true {
	println("");
	println("1. " + option1);
	println("2. " + option2);
	println("3. " + option3);

	var input = int(input("Choose an option: "));
	
	var output = switch_block_options(true, input, result1, result2, result3);
	if output > -1 { return output; }
	}
}

func blank_options(first_piece: string, second_piece: string, option1: string, option2: string, option3: string): int {
	while true {
	println("");
	println(first_piece + "___" + second_piece);
	println("1. " + option1);
	println("2. " + option2);
	println("3. " + option3);

	var input = int(input("Choose an option: "));

	var output = switch_block_options(true, input, first_piece + option1 + second_piece, 
						       first_piece + option2 + second_piece, 
						       first_piece + option3 + second_piece);
	if output > -1 { return output; }
	}
}
