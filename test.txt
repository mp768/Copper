// This is an example of a looping function.
func fib(n: int): int {
    if n <= 1 {
        return n;
    } else {
        // you can use functions at any time.
        return add(fib(n-1), fib(n-2));
    }
}

func add(a: any, b: any): int {
    // will try to convert to an int
    return a + b;
}
